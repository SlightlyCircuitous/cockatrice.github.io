171 sfmt->idx = 0;

174 r1 = psfmt32[sfmt->idx];
175 r2 = psfmt32[sfmt->idx + 1];
176 sfmt->idx += 2;
177 return ((uint64_t)r2 << 32) | r1;
178#else
179 r = psfmt64[sfmt->idx / 2];
180 sfmt->idx += 2;
